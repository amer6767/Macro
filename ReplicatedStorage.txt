local Executor = {}

function Executor.has(funcName)
    return type(getfenv()[funcName]) == "function"
end

function Executor.call(funcName, ...)
    if not Executor.has(funcName) then return nil end
    local success, result = pcall(getfenv()[funcName], ...)
    return success and result or nil
end

function Executor.clipboard(text)
    local fn = Executor.has("setclipboard") and setclipboard 
            or Executor.has("toclipboard") and toclipboard
    if fn then pcall(fn, text) return true end
    return false
end

local UI = {}

function UI.corner(parent, radius)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, radius or 8)
    c.Parent = parent
    return c
end

function UI.stroke(parent, thickness, color, transparency)
    local s = Instance.new("UIStroke")
    s.Thickness = thickness or 1
    s.Color = color or Config.Colors.Accent
    s.Transparency = transparency or 0
    s.Parent = parent
    return s
end

function UI.padding(parent, size)
    local p = Instance.new("UIPadding")
    p.PaddingTop, p.PaddingBottom = UDim.new(0, size), UDim.new(0, size)
    p.PaddingLeft, p.PaddingRight = UDim.new(0, size), UDim.new(0, size)
    p.Parent = parent
    return p
end

function UI.list(parent, padding, direction)
    local l = Instance.new("UIListLayout")
    l.SortOrder = Enum.SortOrder.LayoutOrder
    l.Padding = UDim.new(0, padding or 2)
    l.FillDirection = direction or Enum.FillDirection.Vertical
    l.Parent = parent
    return l
end

function UI.tween(instance, props, duration)
    local ti = TweenInfo.new(duration or 0.12, Enum.EasingStyle.Quart, Enum.EasingDirection.Out)
    game:GetService("TweenService"):Create(instance, ti, props):Play()
end

function UI.frame(props)
    local f = Instance.new("Frame")
    f.BackgroundColor3 = props.Color or Config.Colors.Panel
    f.BackgroundTransparency = props.Transparency or 0
    f.Size = props.Size or UDim2.new(1, 0, 1, 0)
    f.Position = props.Position or UDim2.new(0, 0, 0, 0)
    f.BorderSizePixel = 0
    f.ClipsDescendants = props.Clip or false
    if props.Corner then UI.corner(f, props.Corner) end
    if props.Stroke then UI.stroke(f, props.Stroke.Size, props.Stroke.Color, props.Stroke.Trans) end
    f.Parent = props.Parent
    return f
end

local Toast = { container = nil }

function Toast.init(parent)
    Toast.container = UI.frame({ Size = UDim2.new(0, 320, 1, 0), Position = UDim2.new(1, -330, 0, 0), Transparency = 1, Parent = parent })
    UI.list(Toast.container, 8)
    UI.padding(Toast.container, 15)
end

function Toast.show(message, duration, color)
    if not Toast.container then return end
    
    local toast = UI.frame({ Size = UDim2.new(1, 0, 0, 0), Color = color or Config.Colors.Panel, Corner = 10, Parent = Toast.container })
    toast.AutomaticSize = Enum.AutomaticSize.Y
    UI.stroke(toast, 1, Config.Colors.Accent, 0.5)
    
    local text = UI.label({ Text = message, Size = UDim2.new(1, -16, 0, 0), Position = UDim2.new(0, 8, 0, 8), 
                           Wrap = true, Parent = toast, Bold = true, TextSize = 13 })
    text.AutomaticSize = Enum.AutomaticSize.Y
    
    local pad = UI.frame({ Size = UDim2.new(1, 0, 0, 8), Transparency = 1, Parent = text })
    
    task.delay(duration or 3, function()
        if toast.Parent then
            UI.tween(toast, {BackgroundTransparency = 1}, 0.3)
            UI.tween(text, {TextTransparency = 1}, 0.3)
            task.wait(0.3)
            if toast.Parent then toast:Destroy() end
        end
    end)
end

    if depth > 5 then return "{...}" end
    depth = depth or 1
    local indent = string.rep("    ", depth)
    local parts = {}
    
    for k, v in pairs(tbl) do
        local key = type(k) == "string" and (k:match("^[%a_][%w_]*$") and k or '["'..k..'"]') or "["..tostring(k).."]"
        local val
        
        if type(v) == "string" then
            val = '"' .. v:sub(1, 80):gsub('"', '\\"'):gsub("\n", "\\n") .. (v:len() > 80 and "..." or "") .. '"'
        elseif type(v) == "table" then
            val = serializeTable(v, depth + 1)
        elseif type(v) == "function" then
            val = "function() end"
        else
            val = tostring(v)
        end
        
        table.insert(parts, indent .. key .. " = " .. val)
        if #parts > 25 then table.insert(parts, indent .. "-- ... more entries") break end
    end
    
    if #parts == 0 then return "{}" end
    return "{\n" .. table.concat(parts, ",\n") .. "\n" .. string.rep("    ", depth - 1) .. "}"
end

function Decompiler.decompileModule(moduleScript)
    local ok, module = pcall(require, moduleScript)
    if not ok or module == nil then return nil end
    
    local result = "-- Decompiled ModuleScript (speed's decompiler)\n"
    result = result .. "-- Path: " .. moduleScript:GetFullName() .. "\n\n"
    
    if type(module) == "table" then
        result = result .. "return " .. serializeTable(module, 1)
    elseif type(module) == "function" then
        result = result .. "return function() end -- compiled"
    else
        result = result .. "return " .. tostring(module)
    end
    
    return result
end

function Decompiler.getSource(script)
    local methods = {
        function() return Executor.call("decompile", script) end,
        function() 
            local closure = Executor.call("getscriptclosure", script)
            return closure and Executor.call("decompile", closure)
        end,
        function()
            local ok, src = pcall(function() return gethiddenproperty(script, "Source") end)
            return ok and src and #src > 0 and src
        end,
        function() return Executor.call("getscriptsource", script) end,
        function()
            if script:IsA("ModuleScript") then return Decompiler.decompileModule(script) end
        end,
    }
    
    for _, method in ipairs(methods) do
        local ok, result = pcall(method)
        if ok and result and #result > 10 then
            local lines = 1
            for _ in result:gmatch("\n") do lines = lines + 1 end
            return { source = result, lines = lines, bytes = #result }
        end
    end
    
    -- Fallback info
    local info = "-- üìú " .. script.Name .. " (" .. script.ClassName .. ")\n"
    info = info .. "-- Path: " .. script:GetFullName() .. "\n\n"
    info = info .. "-- Source requires decompiler support\n"
    info = info .. "-- Children: " .. #script:GetChildren()
    
    return { source = info, lines = 5, bytes = #info }
end

local Scanner = {}

function Scanner.getIcon(instance)
    return Config.Icons[instance.ClassName] or 
           (instance:IsA("BaseScript") and Config.Icons.Script) or
           (instance:IsA("BasePart") and Config.Icons.Part) or
           (instance:IsA("Folder") and Config.Icons.Folder) or
           (instance:IsA("Model") and Config.Icons.Model) or
           Config.Icons.Default
end

function Scanner.getColor(instance)
    if instance:IsA("LocalScript") then return Config.Colors.LocalScript
    elseif instance:IsA("Script") then return Config.Colors.Script
    elseif instance:IsA("ModuleScript") then return Config.Colors.ModuleScript
    elseif instance:IsA("Folder") then return Config.Colors.Folder
    else return Config.Colors.Muted end
end

function Scanner.generateMap()
    local maps, totalObjs, totalScripts = {}, 0, 0
    
    for _, serviceName in ipairs(Config.Services) do
        local ok, service = pcall(game.GetService, game, serviceName)
        if not ok or not service then continue end
        
        local lines, objs, scripts = {}, 0, 0
        
        local function scan(inst, depth, path)
            if objs > 1500 then return end
            objs = objs + 1
            if inst:IsA("BaseScript") then scripts = scripts + 1 end
            
            local icon = Scanner.getIcon(inst)
            table.insert(lines, string.rep("  ", depth) .. icon .. " " .. inst.Name .. " [" .. inst.ClassName .. "] -- " .. path)
            
            for _, child in ipairs(inst:GetChildren()) do
                pcall(scan, child, depth + 1, path .. "." .. child.Name)
            end
        end
        
        for _, child in ipairs(service:GetChildren()) do
            pcall(scan, child, 0, serviceName .. "." .. child.Name)
        end
        
        totalObjs, totalScripts = totalObjs + objs, totalScripts + scripts
        maps[serviceName] = { lines = lines, objs = objs, scripts = scripts }
    end
    
    return maps, totalObjs, totalScripts
end

function Scanner.search(query, parent)
    local results = {}
    query = query:lower()
    
    local function searchIn(instance)
        if #results >= Config.Search.MaxResults then return end
        
        for _, child in ipairs(instance:GetChildren()) do
            if #results >= Config.Search.MaxResults then return end
            
            local nameMatch = child.Name:lower():find(query, 1, true)
            local classMatch = child.ClassName:lower():find(query, 1, true)
            
            if nameMatch or classMatch then
                table.insert(results, child)
            end
            
            pcall(searchIn, child)
        end
    end
    
    for _, serviceName in ipairs(Config.Services) do
        if #results >= Config.Search.MaxResults then break end
        local ok, service = pcall(game.GetService, game, serviceName)
        if ok and service then pcall(searchIn, service) end
    end
    
    return results
end

        local maps, totalObjs, totalScripts = Scanner.generateMap()
        if not mapGui.Parent then return end
        loading:Destroy()
        
        local stats = UI.frame({ Size = UDim2.new(1, -20, 0, 32), Position = UDim2.new(0, 10, 0, 62), Color = Config.Colors.Item, Corner = 8, Parent = mapFrame })
        UI.label({ Text = "üìä Total: " .. totalObjs .. " objects | " .. totalScripts .. " scripts", Size = UDim2.new(1, -20, 1, 0), Position = UDim2.new(0, 10, 0, 0), Color = Config.Colors.Accent, Bold = true, Parent = stats })
        
        local tabs = UI.scroll({ Size = UDim2.new(1, -20, 0, 42), Position = UDim2.new(0, 10, 0, 100), Parent = mapFrame })
        tabs.ScrollBarThickness = 0
        tabs.AutomaticCanvasSize = Enum.AutomaticSize.X
        UI.list(tabs, 5, Enum.FillDirection.Horizontal)
        
        local content = UI.scroll({ Size = UDim2.new(1, -20, 1, -155), Position = UDim2.new(0, 10, 0, 148), Parent = mapFrame })
        content.BackgroundColor3 = Config.Colors.Panel
        content.BackgroundTransparency = 0
        UI.corner(content, 10)
        
        local contentBox = Instance.new("TextBox")
        contentBox.Size = UDim2.new(1, -20, 0, 0)
        contentBox.Position = UDim2.new(0, 10, 0, 10)
        contentBox.BackgroundTransparency = 1
        contentBox.TextColor3 = Config.Colors.Text
        contentBox.Font = Enum.Font.Code
        contentBox.TextSize = 12
        contentBox.TextXAlignment = Enum.TextXAlignment.Left
        contentBox.TextYAlignment = Enum.TextYAlignment.Top
        contentBox.TextWrapped = false
        contentBox.MultiLine = true
        contentBox.ClearTextOnFocus = false
        contentBox.TextEditable = false
        contentBox.AutomaticSize = Enum.AutomaticSize.Y
        contentBox.Parent = content
        
        local currentTab = nil
        local tabButtons = {}
        
        local function showService(serviceName)
            if currentTab then currentTab.BackgroundColor3 = Config.Colors.Item end
            if tabButtons[serviceName] then
                tabButtons[serviceName].BackgroundColor3 = Config.Colors.Accent
                currentTab = tabButtons[serviceName]
            end
            
            local data = maps[serviceName]
            if data then
                local header = "\n-- " .. serviceName .. " (" .. data.objs .. " objects, " .. data.scripts .. " scripts)\n\n"
                contentBox.Text = header .. table.concat(data.lines, "\n")
            end
        end
        
        local order = 0
        for _, serviceName in ipairs(Config.Services) do
            if maps[serviceName] and #maps[serviceName].lines > 0 then
                order = order + 1
                local tabBtn = UI.button({ Text = "üìÇ " .. serviceName:sub(1, 12), Size = UDim2.new(0, 120, 0, 34), Color = Config.Colors.Item, TextSize = 11, Corner = 6, Parent = tabs })
                tabBtn.LayoutOrder = order
                tabButtons[serviceName] = tabBtn
                tabBtn.MouseButton1Click:Connect(function() showService(serviceName) end)
                if order == 1 then showService(serviceName) end
            end
        end
        
        local copyAll = UI.button({ Text = "üìã Copy All", Size = UDim2.new(0, 90, 0, 34), Color = Config.Colors.Success, TextSize = 11, Corner = 6, Parent = tabs })
        copyAll.LayoutOrder = 999
        copyAll.MouseButton1Click:Connect(function()
            local fullMap = ""
            for _, sn in ipairs(Config.Services) do
                if maps[sn] then fullMap = fullMap .. "\n-- " .. sn .. "\n" .. table.concat(maps[sn].lines, "\n") .. "\n" end
            end
            Executor.clipboard(fullMap)
            copyAll.Text = "‚úÖ Copied"
            task.wait(1.5)
            copyAll.Text = "üìã Copy All"
        end)
    end)
end
